    (function() {

      class Particle {
        constructor() { this.reset(); }
        reset() {
          this.x = Math.random() * width;
          this.y = -10 - Math.random() * 30;
          this.size = 0.6 + Math.random() * 2.2;
          this.speed = 0.25 + Math.random() * 0.5;
          this.wind = (Math.random() - 0.5) * 0.12;
          this.opacity = 0.1 + Math.random() * 0.25;
          this.isSparkle = Math.random() < 0.2;
          if (this.isSparkle) { this.size = 1.2 + Math.random() * 1.8; this.opacity = 0.3 + Math.random() * 0.4; }
          this.twinkle = Math.random() * Math.PI * 2;
        }
        update() {
          this.y += this.speed * 0.6;
          this.x += this.wind * 0.4 + Math.sin(this.y * 0.005) * 0.1;
          if (this.isSparkle) { this.twinkle += 0.05; this.opacity = 0.3 + Math.sin(this.twinkle) * 0.15 + 0.15; }
          if (this.y > height + 20) { this.reset(); this.y = -10 - Math.random() * 30; }
          if (this.x < -20) this.x = width + 20;
          if (this.x > width + 20) this.x = -20;
        }
        draw(ctx) {
          if (this.isSparkle) {
            ctx.save(); ctx.translate(this.x, this.y); ctx.rotate(this.twinkle);
            const s = this.size * 0.5;
            ctx.beginPath();
            for (let i = 0; i < 4; i++) {
              const angle = (i / 4) * Math.PI * 2;
              ctx.moveTo(0, 0);
              ctx.lineTo(Math.cos(angle) * s, Math.sin(angle) * s);
              ctx.moveTo(0, 0);
              ctx.lineTo(Math.cos(angle + 0.4) * s * 0.4, Math.sin(angle + 0.4) * s * 0.4);
            }
            ctx.strokeStyle = `rgba(255,255,255,${this.opacity * 0.5})`;
            ctx.lineWidth = 0.6; ctx.stroke(); ctx.restore();
          } else {
            ctx.beginPath(); ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
            ctx.fillStyle = `rgba(255,255,255,${this.opacity * 0.5})`;
            ctx.fill();
          }
        }

      const interval = 52;

        if (timeout) clearTimeout(timeout);
        el.innerHTML = ''; idx = 0; active = true;
        if (cursor) cursor.style.display = 'inline-block';
        nextChar();

        if (idx < lyrics.length) {
          const span = document.createElement('span');
          span.className = 'char';
          span.textContent = lyrics.charAt(idx);
          el.appendChild(span);

          all.forEach(c => c.classList.remove('active'));
          if (all.length) {
            const latest = all[all.length-1];

              if (all.length - 1 - i >= 0) all[all.length - 1 - i].style.textShadow = '0 0 8px rgba(255,255,255,0.12)';
            }
          }
          idx++;
          timeout = setTimeout(nextChar, interval);
